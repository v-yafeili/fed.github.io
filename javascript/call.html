# call,apply,bind,new 的实现原理

# call

call 的主要作用：

  • 改变该函数的 this 指向
  • 执行函数

实例

function fn(params) {
	console.log(this);
}
fn.call();
// 在浏览器环境 输出 window
// 在node 环境输出 global

function fn() {
	console.log(1);
}
function fn2() {
	console.log(2);
}
fn.call(fn2); // 输出 1
fn.call.call(fn2); // 输出 2

为什么 fn.call.call(fn2); 输出的是 2 呢？ call 的实现:

Function.prototype.call = function(context) {
	// 首先判断 context 是否存在
	context = context ? Object(context) : global;
	context.fn = this;
	// 处理参数
	let args = [];
	for (let i = 1; i < arguments.length; i++) {
		args.push(`arguments[${i}]`);
	}
	// 执行函数
	let r = eval(`context.fn(${args})`);
	return r;
};

fn.call.call(fn2); 是指 call 执行 ，this 是 fn2 , fn2 执行，输出 2

# apply

apply 和 call 功能类似，call 的值是 1，2，3 ； 而 apply 的参数是素组类型[1,2,3]

fn.apply('hello', [1, 2]);

Function.prototype.apply = function(context, args) {
	// 首先判断 context 是否存在
	context = context ? Object(context) : global;
	context.fn = this;
	if (!args) {
		return context.fn();
	}
	// 执行函数
	let r = eval(`context.fn(${args})`);
	delete context.fn;
	return r;
};

# bind

  • bind 函数可以改变 this 指向
  • bind 返回绑定后的函数
  • 如果返回的函数被 new，当前函数的 this 就是当前实例
  • new 出来的结果，拥有原有类的原型 示例:
let obj = {
	name: '半老师',
};
function fn(name, age) {
	console.log(this.name + '养了一只' + name + age + '岁了');
}
let bindFn = fn.bind(obj, '狗');
bindFn(9); // 半老师养了一只狗9岁了
console.log(bindFn.flag); // undefined

let bind2 = new bindFn(2); //undefined养了一只狗2岁了
console.log(bind2.flag); //哺乳类

bind 实现

Function.prototype.bind = function(context) {
	let that = this;
	// 参数可以多传
	let bindArgs = Array.prototype.slice.call(arguments, 1);
	function Fn() {}
	function fnBind() {
		// 处理调用时的参数
		let args = Array.prototype.slice.call(arguments);
		return that.apply(this instanceof fnBind ? this : context, bindArgs.concat(args));
	}
	Fn.prototype = this.prototype;
	fnBind.prototype = new Fn();
	return fnBind;
};

# new

function Animal(type) {
	this.type = type; // 实例上的属性
}
// 原型上的方法
Animal.prototype.say = function() {
	console.log('say');
};

let animal = new Animal('哺乳类');
console.log(animal.type);
animal.say();
  1. 定义了一个 Animal 类
  2. Animal 的原型(prototype)上定义方法 say
  3. new 处理一个实例 animal
  4. 实例 animal 的 __proto__ 属性指向 Animal 类的原型
function mockNew() {
	// 参数里面第一个为类 剩余的 arguments 就是参数
	let Constructor = [].shift(arguments);
	// 定义实例化类型
	let obj = {};
	// 继承原型上的方法
	obj.__proto__ = Constructor.prototype;
	// 继承实例属性
	let r = Constructor.apply(obj, arguments);
	// 如果构造函数返回是一个对象
	return r instanceof Object ? r : obj;
}
